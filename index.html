async function generateLink() { const headline = document.getElementById('headline').value; const fileInput = document.getElementById('imageUpload'); const linkInput = document.getElementById('generatedLink'); if (!headline || !fileInput.files.length) { alert('Please enter a headline and upload an image.'); return; } const file = fileInput.files[0]; const formData = new FormData(); formData.append('file', file); formData.append('upload_preset', 'wdr_public_uploads'); try { const response = await fetch('https://api.cloudinary.com/v1_1/dgeragc2e/image/upload', { method: 'POST', body: formData }); const data = await response.json(); const previewImageUrl = data.secure_url; const sanitizedHeadline = headline.replace(/\s+/g, '-').toLowerCase(); const previewPath = `news/${sanitizedHeadline}`; // Send metadata to Netlify function to generate a static preview page await fetch('/.netlify/functions/createPreviewPage', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ path: previewPath, image: previewImageUrl, headline: headline }) }); const maskedLink = `https://worlddailyreport.com/${previewPath}`; linkInput.value = maskedLink; } catch (error) { console.error('Error uploading image:', error); alert('Image upload failed.'); } }
